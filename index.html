$os  = Get-CimInstance Win32_OperatingSystem
$cs  = Get-CimInstance Win32_ComputerSystem
$cpu = Get-CimInstance Win32_Processor
$total = [math]::Round($os.TotalVisibleMemorySize / 1MB, 2)
$libre = [math]::Round($os.FreePhysicalMemory / 1MB, 2)
$usada = [math]::Round($total - $libre, 2)
$pct   = [math]::Round(($usada / $total) * 100, 0)
Write-Host "│  OS      : $($os.Caption) — Build $($os.Version)" -ForegroundColor White
Write-Host "│  CPU     : $($cpu.Name.Trim())" -ForegroundColor White

elseif ($pct -gt 55) { Write-Host "${usada} GB usada ($pct%)" -ForegroundColor Yellow }
else { Write-Host "${usada} GB usada ($pct%)" -ForegroundColor Green }
Write-Host "│  Libre   : ${libre} GB" -ForegroundColor White

    @{Path="HKLM:\SOFTWARE\Policies\Microsoft\Windows\DataCollection"; Name="AllowTelemetry";       Expect=0},
    @{Path="HKLM:\SOFTWARE\Policies\Microsoft\Windows\WindowsAI";     Name="DisableAIDataAnalysis"; Expect=1},

Fase 01
Drivers esenciales
Instalar en este orden exacto antes de cualquier otra cosa
1

Chipset AMD

Lo primero. Le dice a Windows cómo hablar con el procesador, la RAM y el resto del hardware.

  • Bajar de amd.com → Drivers & Support
  • Buscar: Ryzen 5 5600G → Chipset
  • Instalar y reiniciar
2

Drivers AMD Adrenalin (GPU)

Instalar en modo Personalizado — nunca Express.

  • Bajar de amd.com → Radeon Graphics
  • Marcar solo: Display Driver y AMD Audio Driver
  • Desmarcar: AMD Link, AMD User Experience, ReLive, Ryzen Master SDK
  • Instalar y reiniciar → ejecutar Bloque 11 (Telemetría AMD) inmediatamente
3

Audio Realtek

Windows generalmente lo instala solo. Si no hay audio, bajar del fabricante de la placa madre. Verificar en Administrador de dispositivos si hay ! amarillo en Audio.

4

Impresora Brother

Conectar por USB antes de instalar el driver.

  • Bajar de support.brother.com
  • Instalar paquete completo (driver + escáner)
5

Verificar dispositivos restantes

Win+X → Administrador de dispositivos. Buscar cualquier ! amarillo y resolver.

$servicios = @(
    "WerSvc","wisvc","RetailDemo","MapsBroker","PhoneSvc",
    "lfsvc","WMPNetworkSvc","icssvc","NcdAutoSetup","MessagingService",
    "TapiSrv","Fax","RemoteRegistry","RasAuto","RasMan",
    "SmsRouter","SNMPTrap","WpcMonSvc","SEMgrSvc","SharedAccess",
    "SCardSvr","ScDeviceEnum","SCPolicySvc","WalletService","WebClient",
    "WbioSrvc","wcncsvc","workfolderssvc","fhsvc","CscService",
    "PeerDistSvc","lmhosts","NetTcpPortSharing","RemoteAccess",
    "UevAgentService","autotimesvc","MSiSCSI","MsKeyboardFilter",
    "dot3svc","EapHost","KtmRm","MSDTC","RpcLocator",
    "WinRM","ssh-agent","tzautoupdate","RmSvc","WwanSvc",
    "McmSvc","perceptionsimulation"
)

Write-Host "Iniciando deshabilitación de servicios..." -ForegroundColor Cyan
Write-Host "=======================================" -ForegroundColor Cyan

foreach ($s in $servicios) {
    try {
        $svc = Get-Service -Name $s -ErrorAction Stop
        
        if ($svc.StartType -eq 'Disabled') {
            Write-Host " ✓ $s ya estaba deshabilitado" -ForegroundColor DarkGreen
            continue
        }

        Set-Service -Name $s -StartupType Disabled -ErrorAction Stop
        Stop-Service -Name $s -Force -ErrorAction SilentlyContinue
        
        Write-Host " ✔ $s → Deshabilitado y detenido" -ForegroundColor Green
    }
    catch [Microsoft.PowerShell.Commands.ServiceCommandException] {
        Write-Host " ⚠ No encontrado: $s" -ForegroundColor DarkGray
    }
    catch {
        Write-Host " ✘ Error en $s : $($_.Exception.Message)" -ForegroundColor Red
    }
}

Write-Host "`n ► Bloque 1 completado." -ForegroundColor Cyan
Write-Host "   Reinicia el PC para aplicar los cambios completamente." -ForegroundColor Yellow
